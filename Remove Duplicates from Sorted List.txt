 
 Remove Duplicates from Sorted List
 
 
 class Solution {
    public ListNode deleteDuplicates(ListNode head) {
        if(head ==null) return null;
        
        Set s = new HashSet<>();
        s.add(head.val);
        ListNode newh = new ListNode(head.val);
        ListNode n = newh;
        
        while(head!=null){
            if(s.add(head.val)){
                n.next = new ListNode(head.val);
                n=n.next;
            }
             head = head.next;
        }                
        return newh;
        
    }
}
 
 
 public ListNode deleteDuplicates(ListNode head) {
        if(head == null) return null;
        ListNode newHead = head;
        ListNode prev = head;
        ListNode cur = newHead;  //cur of new linkedlist 
        head = head.next;
        
        while (head != null){
            if(head.val != prev.val){
                cur.next = head;
                prev = head;
                cur = cur.next;
            }
            head = head.next;  
        }
        cur.next = head;
        return newHead;
    }
